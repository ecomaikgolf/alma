41 /* Load the kernel */
42 Elf64_Ehdr *elf_header = load_elf("kernel.elf");

44 if (elf_header == NULL) {
45 error("cannot load the kernel");

49 /* Retrieve the Graphics Output Protocol */
50 efi_gop_t *gop = load_gop();

60 if (gop == NULL) {

63 }
64
65 /**
66 * Load UEFI memory map
67 *
68 * @warning Moving this function to other line breaks kernel printing
69 */
70 MapInfo *map = load_memmap();
71
72 /* Obtain the RSDP */
73 rsdp_v2 *rsdp = load_rsdp();
74
75 /* Load the PSF1 font */
76 PSF1_Font *font = load_psf1_font("zap-light16.psf");
77
78 if (font == NULL) {
79 error("font load failure");
80 return PSF1_FAILURE;
81 }
82
83 /* Check for suspicious values leading to errors */
84 if (elf_header->e_entry == 0x0)
85 warning("kernel entry point is 0x0");
86 if (elf_header->e_entry < 0x100)
87 warning("kernel entry point is less than 0x100");
88
89 /* Define the kernel starting function
90 * sysv_abi as UEFI uses cdecl for x86 and MS 64-bit convention for x86-64
91 */
92 void (*_start)() = ((__attribute__((sysv_abi)) void (*)(BootArgs *))elf_header->e_entry);
93 info("jumping to kernel code at address: 0x%p", _start);
94
95 info("calling kernel global constructors");
96
97 call_ctors(elf_header);
98
99 info("finished kernel global constructors");
100
101 /* Exit UEFI Boot Services */
102 // info("Exiting UEFI Boot Services before the jump");
103 // if (exit_bs() > 0) {
104 // error("error exiting UEFI boot services");
105 // return UEFI_BS;
106 // }
107
108 /* Call the kernel */
109 BootArgs args = { fb, font, map, rsdp };
110 _start(&args);
111
112 /* _start shouldn't return */
113 while (1) {
114 }
115
116 return 0;
117}
