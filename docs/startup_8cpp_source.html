66void
67interrupts()
68{
69 /* Locks the IDT */
70 kernel::allocator.lock_pages(&kernel::idtr, sizeof(kernel::idtr) / kernel::page_size + 1);
71 kernel::idtr.set_ptr((uint64_t)kernel::allocator.request_page());
72
73 /* Load the interrupt handlers */
74 kernel::idtr.add_handle(interrupts::vector_e::reserved, interrupts::reserved);
75
76 /*
77 * From OSDev:
78 * In protected mode, the IRQs 0 to 7 conflict with the CPU exception which are reserved
79 * by Intel up until 0x1F. It is thus recommended to change the PIC's offsets (also known
80 * as remapping the PIC) so that IRQs use non-reserved vectors. A common choice is to move
81 * them to the beginning of the available range (IRQs 0..0xF -> INT 0x20..0x2F). For that,
82 * we need to set the master PIC's offset to 0x20 and the slave's to 0x28.
83 */
84 kernel::idtr.remap_pic(0x20, 0x28);
85
86 io::outb(io::PIC1_DATA, 0b11111101); // enable IRQ1 from PIC1 (keyboard)
87 io::outb(io::PIC2_DATA, 0b11111111); // mask all lines
88
89 /* Enable interrupts */
90 asm("sti");
91}
