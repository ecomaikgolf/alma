11 *
12 * @param filename Name of the file
13 * @return Font struture

15PSF1_Font *
16load_psf1_font(const char *const filename)
17{
18 /* Copy file contents to memory */
19 char *memory = load_file(filename);
20
21 /* Not enough memory */
22 if (memory == NULL)
23 return NULL;
24
25 /* Get PSF1 header struct */
26 PSF1_Header *header = get_psf1_header(memory);
27
28 /* Verify PSF1 header */
29 uint8_t flags = verify_psf1_header(header);
30
31 /* Incorrect header */
32 if (flags > 0)
33 return NULL;
34
35 /* Get the glyph font data
36 * https://github.com/ecomaikgolf/os-dev/raw/master/toolchain/font/zap-vga16.pdf
37 */
38 void *glpyh_buffer = get_psf1_glyph(memory);
39
40 /* Create the font structure */
41 PSF1_Font *font = malloc(sizeof(PSF1_Font));
42
43 /* Not enough memory */
44 if (font == NULL)
45 return NULL;
46
47 font->header = header;
48 font->buffer = glpyh_buffer;
49
50 return font;
51}

67/**
68 * Verify PSF1 header magic numbers
69 *
70 * 0x36, 0x04 at the start of the header
71 *
72 * @param header Header to check
73 * @return Flags of the check, 1 if incorrect, 0 if correct
74 */
75uint8_t
76verify_psf1_header(const PSF1_Header *const header)
77{
78 /* magic[0] == 0x36, magic[1] == 0x04 */
79 if (header->magic[0] != PSF1_MAGIC0 || header->magic[1] != PSF1_MAGIC1) {
80 error("PSF1 font incorrect magic header");
81 return 1;
82 } else {
83 info("PSF1 font correct magic header");
84 return 0;
85 }
86}
