46/*
47 * Extended Numbering
48 *
49 * If the real number of program header table entries is larger than
50 * or equal to PN_XNUM(0xffff), it is set to sh_info field of the
51 * section header at index 0, and PN_XNUM is set to e_phnum
52 * field. Otherwise, the section header at index 0 is zero
53 * initialized, if it exists.
54 *
55 * Specifications are available in:
56 *
57 * - Oracle: Linker and Libraries.
58 * Part No: 817–1984–19, August 2011.
59 * https://docs.oracle.com/cd/E18752_01/pdf/817-1984.pdf
60 *
61 * - System V ABI AMD64 Architecture Processor Supplement
62 * Draft Version 0.99.4,
63 * January 13, 2010.
64 * http://www.cs.washington.edu/education/courses/cse351/12wi/supp-docs/abi.pdf
65 */
66#define PN_XNUM 0xffff
