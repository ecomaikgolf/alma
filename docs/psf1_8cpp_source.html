47/**
48 * Draw a character to the screen
49 *
50 * Doesn't increase x_offset and y_offset, also does not recognise special chars
51 *
52 * Overrided from renderer interface
53 *
54 * @param character Char to print
55 */
56void
57psf1_renderer::draw(const char character)
58{
59 /* Select character from glyph buffer (a character is composed by charsize
60 * elements )*/
61 char *chr = static_cast<char *>(this->font->buffer) +
62 (static_cast<unsigned char>(character) * this->font->header->charsize);
63
64 /* Iterate bitmap "rectangle" (with offset as base) */
65 for (unsigned long y = this->y_offset; y < this->y_offset + this->glyph_y(); y++) {
66 for (unsigned long x = this->x_offset; x < this->x_offset + this->glyph_x(); x++) {
67 /* Each Y from bitmap is a "flag number", check if corresponding
68 * x bit is set */
69 if ((*chr & (0b10000000 >> (x - this->x_offset)))) {
70 *(static_cast<unsigned int *>(this->fb->base + x + (y * this->fb->ppscl))) =
71 static_cast<unsigned int>(this->color);
72 }
73 }
74 chr++; /* increase char to iterate over entire charsize elements */
75 }
76}
