28/**
29 * Print a string without default newline
30 *
31 * Increases x_offset and y_offset, also recognises special chars as '\n'
32 *
33 * @param str string to print
34 */
35void
36renderer_i::print(const char *str)
37{
38 int i = 0;
39 while (str[i]) {
40 switch (str[i]) {
41 case '\n':
42 this->y_offset += this->glyph_y();
43 this->x_offset = 0;
44 break;
45 default: {
46 this->draw(str[i]);
47 this->x_offset += this->glyph_x();
48 if (this->x_offset >= this->fb->width)
49 this->print("\n");
50 }
51 }
52 if ((this->y_offset + this->glyph_y()) > this->fb->height)
53 this->scroll();
54 i++;
55 }
56}

94/**
95 * Fast screen clear
96 *
97 * Uses biggest integer asignations
98 */
99void
100renderer_i::clear()
101{
102 uint32_t jumps = fb->buffer_size / sizeof(uint64_t);
103 uint32_t rest = fb->buffer_size % sizeof(uint64_t);
104
105 /* Fast clear, use big integer movements */
106 unsigned int i;
107 for (i = 0; i < jumps; i++)
108 *((uint64_t *)fb->base + i) = static_cast<uint8_t>(screen::color_e::BLACK);
109
110 /* for sizes < 64 bytes */
111 for (unsigned int j = 0; j < rest; j++)
112 *((uint8_t *)((uint64_t *)fb->base + i) + j) = static_cast<uint8_t>(screen::color_e::BLACK);
113
114 this->x_offset = 0;
115 this->y_offset = 0;
116}

118void
119renderer_i::scroll()
120{
121 uint32_t mrg = fb->ppscl * (this->glyph_y() * 4);
122 uint32_t dif = fb->buffer_size - mrg;
123 uint8_t *src = (uint8_t *)fb->base + mrg;
124 uint8_t *dst = (uint8_t *)fb->base;
125
126 {
127 uint32_t jumps = dif / sizeof(uint64_t);
128 uint32_t rest = dif % sizeof(uint64_t);
129
130 unsigned int i;
131 for (i = 0; i < jumps; i++)
132 *((uint64_t *)dst + i) = *((uint64_t *)src + i);
133
134 for (unsigned int j = 0; j < rest; j++)
135 *((uint8_t *)((uint64_t *)dst + i) + j) = *((uint8_t *)((uint64_t *)src + i) + j);
136 }
137
138 {
139 uint8_t *clr = ((uint8_t *)fb->base + fb->buffer_size) - mrg;
140
141 uint32_t jumps = mrg / sizeof(uint64_t);
142 uint32_t rest = mrg % sizeof(uint64_t);
143
144 unsigned int i;
145 for (i = 0; i < jumps; i++)
146 *((uint64_t *)clr + i) = 0;
147
148 for (unsigned int j = 0; j < rest; j++)
149 *((uint8_t *)((uint64_t *)clr + i) + j) = 0;
150 }
151
152 if (this->y_offset >= this->glyph_y())
153 this->y_offset -= this->glyph_y();
154}
