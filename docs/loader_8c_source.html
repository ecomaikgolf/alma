43/**
44 * Load ELF file from a memory chunk
45 *
46 * Parses, verifies, does the memory management, etc.
47 */
48Elf64_Ehdr *
49load_elf(const char *const filename)
50{
51 char *memory = (char *)load_file(filename);
52
53 if (memory == NULL)
54 return NULL;
55
56 Elf64_Ehdr *header = get_elf_header(memory);
57 uint8_t flags = verify_elf_headers(header);
58
59 if (flags > 0) {
60 error("wrong ELF header, status code: %d", flags);
61 return NULL;
62 }
63
64 load_phdrs(header, memory);
65
66 return header;
67}

81/**
82 * Verify ELF header
83 *
84 * Performs switch defined checks to see if the header is correct and good to load
85 *
86 * To add more checks:
87 * - Add a new case. Mark check 0 if correct, 1 if not correct
88 * - Append a new info string and error string in the header
89 * - Increase NUM_CHECKS in the header by one
90 * - Max 8 checks, increase return int size to add more flags
91 */
92uint8_t
93verify_elf_headers(const Elf64_Ehdr *const elf_header)
94{
95 if (elf_header == NULL) {
96 warning("elf_header parameter is null");
97 return 255;
98 }
99
100 uint8_t elf_parse_flags = 0;
101
102 for (int i = 0; i < NUM_CHECKS; i++) {
103 uint8_t check = 255; /* NUM_CHECKS > actual checks, throw warning */
104 switch (i) {
105 case 0:
106 /* Magic header */
107 check = !(memcmp(elf_header->e_ident, ELFMAG, SELFMAG) == 0);
108 break;
109 case 1:
110 /* Executable ELF */
111 check = !(elf_header->e_type == ET_EXEC);
112 break;
113 case 2:
114 /* ELF Architecture */
115 check = !(elf_header->e_machine == EM_MACH);
116 break;
117 case 3:
118 /* ELF 64 bits target */
119 check = !(elf_header->e_ident[EI_CLASS] == ELFCLASS64);
120 break;
121 case 4:
122 /* ELF non empty program headers */
123 check = !(elf_header->e_phnum > 0);
124 break;
125 default:
126 printf("(W) [bootloader] verify_elf_headers(...) default switch case reached\n");
127 }
128
129 if (check != 255) {
130 /* Mark the bit with 1 to identify the check error */
131 elf_parse_flags |= check << i;
132
133 (check > 0) ? error("%s", errormessages[i]) : info("%s", verifymessages[i]);
134 }
135 }
136
137 return elf_parse_flags;
138}

140/**
141 * Load ELF program segments
142 *
143 * @post Loaded program headers are in memory
144 */
145void
146load_phdrs(const Elf64_Ehdr *const elf_header, const char *const memory)
147{
148 for (int i = 0; i < elf_header->e_phnum; i++) {
149 Elf64_Phdr *prog_hdr =
150 (Elf64_Phdr *)(memory + elf_header->e_phoff + elf_header->e_phentsize * i);
151 switch (prog_hdr->p_type) {
152 /* Loadable */
153 case PT_LOAD: {
154 info("loading program header %d at: 0x%p", i, prog_hdr->p_vaddr);
155
156 memcpy((void *)prog_hdr->p_vaddr, memory + prog_hdr->p_offset, prog_hdr->p_filesz);
157 memset((void *)(prog_hdr->p_vaddr + prog_hdr->p_filesz),
158 0,
159 prog_hdr->p_memsz - prog_hdr->p_filesz);
160 break;
161 }
162 /* Ignore */
163 default:
164 info("program header %d of type %d ignored", i, prog_hdr->p_type);
165 }
166 }

169void
170call_ctors(Elf64_Ehdr *elf)
171{
172 /* e->shentsize is the index of the string table section */
173 Elf64_Shdr *str_table_hdr =
174 (Elf64_Shdr *)((char *)elf + elf->e_shoff + (elf->e_shentsize * elf->e_shstrndx));
175
176 char *str_table = ((char *)elf + str_table_hdr->sh_offset);
177
178 /* Iterate over section headers, skip the first as it's the NULL section header */
179 for (int i = 1; i < elf->e_shnum; i++) {
180 Elf64_Shdr *header = (Elf64_Shdr *)((char *)elf + elf->e_shoff + elf->e_shentsize * i);
181
182 /* http://www.sco.com/developers/gabi/2003-12-17/ch4.strtab.html */
183 char *section_name = str_table + header->sh_name;
184
185 if (strncmp(section_name, ".ctors", 7) == 0) {
186 uint64_t *ctors = (uint64_t *)((char *)elf + header->sh_offset);
187
188 /* / uint64_t as 64bit systems have 64bit addr */
189 uint64_t num_functions = (header->sh_size / sizeof(uint64_t));
190
191 for (uint64_t i = 0; i < num_functions; i++) {
192 /* sysv_abi as we are on a PE executable */
193 void (*func)() = ((__attribute__((sysv_abi)) void (*)(void)) * ctors);
194 func();
195 ctors++;
196 }
197 }
198 }
